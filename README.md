## Install

```bash
npm install --save react-leaflet-component
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-leaflet-component'
import 'react-leaflet-component/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
